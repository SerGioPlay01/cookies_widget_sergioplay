üîí Script Guard

Script blocking for Analytics and Marketing even before site loading through proxy stubs.

How it works

üõ°Ô∏è Protection

Script Guard intercepts tracker function calls (gtag, fbq, ym) and blocks their execution until user consent is obtained.

// Script Guard automatically blocks:
gtag('config', 'GA_MEASUREMENT_ID'); // ‚ùå Blocked
fbq('init', 'PIXEL_ID');             // ‚ùå Blocked  
ym(12345678, "init", {});            // ‚ùå Blocked

// After consent is given:
gtag('config', 'GA_MEASUREMENT_ID'); // ‚úÖ Executed
fbq('init', 'PIXEL_ID');             // ‚úÖ Executed

üìä Google Analytics 4 - Complete Integration

JavaScript 
// Advanced Google Analytics 4 integration
class GA4CookieIntegration {
    constructor(measurementId) {
        this.measurementId = measurementId;
        this.isLoaded = false;
        this.init();
    }
    
    init() {
        // Listen to cookie banner events
        window.addEventListener('cookieConsentAccepted', (event) => {
            if (event.detail.analytics) {
                this.loadGA4();
                this.trackConsentAccepted(event.detail);
            }
        });
        
        window.addEventListener('cookieConsentDeclined', (event) => {
            this.trackConsentDeclined();
            this.disableGA4();
        });
        
        window.addEventListener('cookieConsentChanged', (event) => {
            if (event.detail.analytics && !this.isLoaded) {
                this.loadGA4();
            } else if (!event.detail.analytics && this.isLoaded) {
                this.disableGA4();
            }
        });
    }
    
    loadGA4() {
        if (this.isLoaded) return;
        
        // Load Google Analytics 4
        const script = document.createElement('script');
        script.async = true;
        script.src = `https://www.googletagmanager.com/gtag/js?id=${this.measurementId}`;
        document.head.appendChild(script);
        
        window.dataLayer = window.dataLayer || [];
        function gtag(){dataLayer.push(arguments);}
        window.gtag = gtag;
        
        gtag('js', new Date());
        gtag('config', this.measurementId, {
            anonymize_ip: true,
            allow_google_signals: false,
            allow_ad_personalization_signals: false
        });
        
        this.isLoaded = true;
        console.log('Google Analytics 4 loaded');
    }
    
    disableGA4() {
        if (!this.isLoaded) return;
        
        // Disable GA4
        window[`ga-disable-${this.measurementId}`] = true;
        console.log('Google Analytics 4 disabled');
    }
    
    trackConsentAccepted(settings) {
        if (window.gtag) {
            gtag('event', 'cookie_consent_accepted', {
                'custom_parameter': JSON.stringify(settings)
            });
        }
    }
    
    trackConsentDeclined() {
        if (window.gtag) {
            gtag('event', 'cookie_consent_declined');
        }
    }
}

// Initialize
const ga4Integration = new GA4CookieIntegration('GA_MEASUREMENT_ID');

// Yandex.Metrica integration with 152-FZ compliance
class YandexMetricaCookieIntegration {
    constructor(counterId) {
        this.counterId = counterId;
        this.isLoaded = false;
        this.init();
    }
    
    init() {
        window.addEventListener('cookieConsentAccepted', (event) => {
            if (event.detail.analytics) {
                this.loadYandexMetrica();
                this.trackConsentEvent('accepted', event.detail);
            }
        });
        
        window.addEventListener('cookieConsentDeclined', (event) => {
            this.trackConsentEvent('declined');
        });
    }
    
    loadYandexMetrica() {
        if (this.isLoaded) return;
        
        // Load Yandex.Metrica
        (function(m,e,t,r,i,k,a){
            m[i]=m[i]||function(){(m[i].a=m[i].a||[]).push(arguments)};
            m[i].l=1*new Date();
            k=e.createElement(t),a=e.getElementsByTagName(t)[0],
            k.async=1,k.src=r,a.parentNode.insertBefore(k,a)
        })(window, document, "script", "https://mc.yandex.ru/metrika/tag.js", "ym");
        
        ym(this.counterId, "init", {
            clickmap: true,
            trackLinks: true,
            accurateTrackBounce: true,
            webvisor: false, // Disable webvisor for privacy compliance
            trackHash: true,
            ecommerce: "dataLayer"

                settings: JSON.stringify(settings)
            });
        }
    }
}

// components/CookieManager.tsx
import { useEffect, useState } from 'react';
import Head from 'next/head';

interface CookieSettings {
    necessary: boolean;
    functional: boolean;
    analytics: boolean;
    marketing: boolean;
    performance: boolean;
}

export default function CookieManager() {
    const [cookieSettings, setCookieSettings] = useState<CookieSettings | null>(null);
    const [isInitialized, setIsInitialized] = useState(false);
    
    useEffect(() => {
        // Initialize cookie banner
        const initCookieBanner = () => {
            if (typeof window !== 'undefined' && window.CookieBanner) {
                const banner = new window.CookieBanner({
                    autoShow: true,
                    theme: 'modern',
                    position: 'bottom',
                    privacyPolicyUrl: '/privacy-policy',
                    onAccept: (settings: CookieSettings) => {
                        setCookieSettings(settings);
                        handleCookieAccept(settings);
                    },
                    onDecline: (settings: CookieSettings) => {
                        setCookieSettings(settings);
                        handleCookieDecline(settings);
                    }
                });
                
                setIsInitialized(true);
                
                // Customization for React application
                banner.setTexts({
                    title: 'Data Management',
                    description: 'We use modern technologies to improve your experience',
                    acceptAll: 'Accept All',
                    decline: 'Necessary Only',
                    settings: 'Configure'
                });
            }
        };
        
        // Check script loading
        if (document.readyState === 'complete') {
            initCookieBanner();
        } else {
            window.addEventListener('load', initCookieBanner);
        }
        
        return () => {
            window.removeEventListener('load', initCookieBanner);
        };
    }, []);
    
    const handleCookieAccept = (settings: CookieSettings) => {
        // Load analytics if allowed
        if (settings.analytics) {
            import('../utils/analytics').then(({ initAnalytics }) => {
                initAnalytics();
            });
        }
        
        // Load marketing scripts
        if (settings.marketing) {
            import('../utils/marketing').then(({ initMarketing }) => {
                initMarketing();
            });
        }
    };
    
    const handleCookieDecline = (settings: CookieSettings) => {
        console.log('User declined cookies:', settings);
    };
    
    return (
        <>
            <Head>
                <link 
                    rel="stylesheet" 
                    href="https://cookies-widget-sergioplay.vercel.app/cookie-banner.css" 
                />
                <script 
                    src="https://cookies-widget-sergioplay.vercel.app/cookie-banner.js"
                    async
                />
            </Head>
            
            {/* Custom settings button in footer */}
            {isInitialized && (
                <button 
                    onClick={() => window.cookieBanner?.show()}
                    className="cookie-settings-btn"
                >
                    üç™ Cookie Settings
                </button>
            )}
        </>
    );
}

// Smart consent-based script loading system
class SmartScriptLoader {
    constructor() {
        this.loadedScripts = new Set();
        this.scriptQueue = new Map();
        this.init();
    }
    
    init() {
        // Register scripts for each category
        this.registerScripts();
        
        // Listen to cookie banner events
        window.addEventListener('cookieConsentAccepted', (event) => {
            this.loadScriptsForCategories(event.detail);
        });
        
        window.addEventListener('cookieConsentChanged', (event) => {
            this.handleConsentChange(event.detail);
        });
    }
    
    registerScripts() {
        // Analytics scripts
        this.addScript('analytics', {
            name: 'Google Analytics',
            src: 'https://www.googletagmanager.com/gtag/js?id=GA_MEASUREMENT_ID',
            init: () => {
                window.dataLayer = window.dataLayer || [];
                function gtag(){dataLayer.push(arguments);}
                gtag('js', new Date());
                gtag('config', 'GA_MEASUREMENT_ID');
            }
        });
        
        this.addScript('analytics', {
            name: 'Yandex Metrica',
            src: 'https://mc.yandex.ru/metrika/tag.js',
            init: () => {
                ym(12345678, "init", {
                    clickmap: true,
                    trackLinks: true,
                    accurateTrackBounce: true
                });
            }
        });
        
        // Marketing scripts
        this.addScript('marketing', {
            name: 'Facebook Pixel',
            src: 'https://connect.facebook.net/en_US/fbevents.js',
            init: () => {
                fbq('init', 'YOUR_PIXEL_ID');
                fbq('track', 'PageView');
            }
        });
        
        // Functional scripts
        this.addScript('functional', {
            name: 'Chat Widget',
            src: 'https://widget.intercom.io/widget/YOUR_APP_ID',
            init: () => {
                window.Intercom('boot', {
                    app_id: 'YOUR_APP_ID'
                });
            }
        });
    }
    
    addScript(category, config) {
        if (!this.scriptQueue.has(category)) {
            this.scriptQueue.set(category, []);
        }
        this.scriptQueue.get(category).push(config);
    }
    
    loadScriptsForCategories(settings) {
        Object.keys(settings).forEach(category => {
            if (settings[category] && this.scriptQueue.has(category)) {
                this.loadCategoryScripts(category);
            }
        });
    }
    
    loadCategoryScripts(category) {
        const scripts = this.scriptQueue.get(category) || [];
        
        scripts.forEach(scriptConfig => {
            const scriptId = `${category}-${scriptConfig.name}`;
            
            if (!this.loadedScripts.has(scriptId)) {
                this.loadScript(scriptConfig, scriptId);
            }
        });
    }
    
    loadScript(config, scriptId) {
        const script = document.createElement('script');
        script.src = config.src;
        script.async = true;
        script.onload = () => {
            console.log(`Script loaded: ${config.name}`);
            if (config.init) {
                config.init();
            }
            this.loadedScripts.add(scriptId);
        };
        script.onerror = () => {
            console.error(`Script loading error: ${config.name}`);
        };
        
        document.head.appendChild(script);
    }
    
    handleConsentChange(newSettings) {
        // Load new scripts if consent expanded
        this.loadScriptsForCategories(newSettings);
        
        // Here you can add logic to disable scripts
        // if consent was revoked
    }
}
